			tag = NHS

		owns = 2554 # Nanticoke

		controls = 953
		controls = 952
		controls = 950
		controls = 2550
		controls = 2554
	}

		months = 1
	}

	option = {
		name = "NHSCOUNTRYEVENTSNAME2"
		add_years_of_income = 0.2

		# Rename Chesapeake
		950 = {
			rename_capital = "Didymo Fylakes"

				name = "nhs_toll_modifier"
				duration = -1
			}
			set_province_flag = nhs_original_heartland
			add_base_tax = 1
			add_base_production = 4
		}
		hidden_effect = {
			952 = {
				set_province_flag = nhs_original_heartland

					limit = {
						NOT = { has_province_flag = got_base_tax_new }

					add_base_production = 1

			}

				set_province_flag = nhs_original_heartland

					add_base_production = 1
					set_province_flag = got_base_tax_new
				}
			}
			2554 = { 
				set_province_flag = nhs_original_heartland

					add_base_production = 1

				}

			950 = {
				set_province_flag = nhs_original_heartland

					limit = {
						NOT = { has_province_flag = got_base_tax_new }
					}

# Allow second wives and concubines, ally with a native faction
country_event = {
	id = nhs_countryevents.4
	title = nhs_countryevents.4.t
	desc = nhs_countryevents.4.d
	picture = COURT_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		OR = {
			tag = NHS
			tag = ELY
			tag = MSN
			tag = MSG
		}
		is_year = 1460
		religion = pantheon_worship
		any_neighbor_country = {
			religion_group = pagan
			is_at_war = no
			has_opinion = { who = ROOT value = -50 }
			NOT = { is_subject_of = ROOT }
			NOT = { alliance_with = ROOT }
		}
		NOT = { has_country_modifier = nhs_allows_polygamy }
	}

	mean_time_to_happen = {
		months = 12
	}

	option = {		# Allow for it
		name = "NHSCOUNTRYEVENTSNAME4A"
		ai_chance = { 
			factor = 60
			modifier = {
				factor = 2
				legitimacy = 95
			}
			modifier = {
				factor = 0.2
				NOT = {
					legitimacy = 60
				}
			}
		}
		add_legitimacy = -15
		add_country_modifier = {
			name = "nhs_allows_polygamy"
			duration = -1
		}
		random_neighbor_country = {
			limit = {
				religion_group = pagan
				is_at_war = no
				has_opinion = { who = ROOT value = -50 }
				NOT = { is_subject_of = ROOT }
				NOT = { alliance_with = ROOT }
			}
			create_alliance = ROOT
			if = {
				limit = {
					OR = {
						government = despotic_monarchy
						government = feudal_monarchy
						government = administrative_monarchy
						government = noble_republic
						government = constitutional_monarchy
						government = enlightened_despotism
						government = steppe_horde
						government = tribal_despotism
						government = tribal_federation
					}
				}
				create_marriage = ROOT
			}
			add_opinion = {
				who = ROOT
				modifier = nhs_accepted_gift
			}
			reverse_add_opinion = {
				who = ROOT
				modifier = nhs_accepted_gift
			}
		}
	}
	option = {		# Reject
		name = "NHSCOUNTRYEVENTSNAME4B"
		ai_chance = { 
			factor = 40
			modifier = {
				factor = 0.5
				legitimacy = 95
			}
			modifier = {
				factor = 2
				NOT = {
					legitimacy = 60
				}
			}
		}
		add_legitimacy = 10
	}
}

# Natives discover Gold on our borders
country_event = {
	id = nhs_countryevents.9
	title = nhs_countryevents.9.t
	desc = nhs_countryevents.9.d
	picture = MINE_2_eventPicture

	fire_only_once = yes

    trigger = {
		OR = {
			tag = NHS
			tag = ELY
		}
		is_year = 1500
		any_owned_province = {
			any_neighbor_province = {
				NOT = { trade_goods = gold }
				is_empty = no
				NOT = { is_claim = ROOT }
				religion_group = pagan
				owner = { 
					NOT = { is_subject_of = ROOT }
					NOT = { alliance_with = ROOT }
					NOT = { tag = NHS }
					NOT = { tag = ELY }
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 1000
		modifier = {
			factor = 0.75
			any_owned_province = {
				any_neighbor_province = {
					NOT = { trade_goods = gold }
					is_empty = no
					NOT = { is_claim = ROOT }
					religion_group = pagan
					owner = { 
						NOT = { is_subject_of = ROOT }
						NOT = { alliance_with = ROOT }
						NOT = { tag = NHS }
						NOT = { tag = ELY }
						NOT = { has_opinion = { who = ROOT value = 0 } }
					}
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					any_neighbor_province = {
						NOT = { trade_goods = gold }
						is_empty = no
						NOT = { is_claim = ROOT }
						religion_group = pagan
						owner = { 
							NOT = { is_subject_of = ROOT }
							NOT = { alliance_with = ROOT }
							NOT = { tag = NHS }
							NOT = { tag = ELY }
						}
					}
				}
				random_neighbor_province = {
					limit = {
						NOT = { trade_goods = gold }
						is_empty = no
						NOT = { is_claim = ROOT }
						religion_group = pagan
						owner = { 
							NOT = { tag = NHS }
							NOT = { tag = ELY }
							NOT = { is_subject_of = ROOT }
							NOT = { alliance_with = ROOT }
						}
					}
					set_province_flag = nhs_province_gold
				}
			}
		}
	}

	option = {
		name = "NHSCOUNTRYEVENTSNAME9"
		random_province = {
			limit = {
				has_province_flag = nhs_province_gold
			}
			change_trade_goods = gold
			add_claim = ROOT
			owner = {
				add_opinion = {
					who = ROOT
					modifier = nhs_borderconflict_gold
				}
				reverse_add_opinion = {
					who = ROOT
					modifier = nhs_borderconflict_gold
				}
			}
			clr_province_flag = nhs_province_gold
		}
	}
}
